2.命名规范:

根据每块元素的主题 或者功能、在页面上的位置命名，每个单词中间以“_”隔开或者从第二个单词开始每个单词的首字母大写（写包含样式的时候能找到这个元素并且不影响其他元素即可）

登录条：loginBar         标志：logo              侧栏：sideBar        
子导航subNav              广告：banner            导航：nav

滚动：scroll             页面主体：main          内容：content

CSS

1.样式重置：

做到有标签就加否则就不加，勿滥用通配选择符*

`body,h1,h2,h3,h4,h5,h6,p,dl,dd,ul,ol,form,input,textarea,th,td,select{
    margin:0;
    padding: 0;
}
em{ font-style:normal; }
li{list-style:none;}
a{ text-decoration:none; }
img{ border:none; vertical-align:top; }
table{ border-collapse:collapse; } 表格的两边框合并为一条
input,textarea{ outline:none; }  去掉chrome 浏览器 input textarea 获取焦点后边框
textarea{ resize: none; overflow: auto; } 关闭 textarea手动拖拉改变长宽和右侧滚动条
body{ font-size:; font-family:""; }
`

2.清除浮动：

`.clear{ zoom:1;/*有宽能触发zoom就不用加*/ }
 .clear:after{ content: ''; display: table; clear: both; };`

3.绝对定位居中：

`position: absolute;left: 50%; top: ; margin-left:-自身宽度的一半`

4.定位position：

absolute static relative fixed inherit

position值为absolute：

`该元素将变为块级元素，相当于给该元素设置了 display: block;（给一个内联元素，如 <span> ，设置 absolute 之后发现它可以设置宽高了）`
`如果该元素是块级元素，元素的宽度由原来的 width: 100%（占据一行），变为了 auto`

2.轮播图

    `thaiq.app.toBanner=function(){
    var oAd=document.getElementById('ad');
    var aLi=oAd.getElementsByTagName('li');

    var oPrevBg=thaiq.tools.getByClass(oAd,'prev_bg')[0];
    var oNextBg=thaiq.tools.getByClass(oAd,'next_bg')[0];

    var oPrev=thaiq.tools.getByClass(oAd,'prev')[0];
    var oNext=thaiq.tools.getByClass(oAd,'next')[0];

    var iNow=0;
    var timer=setInterval(auto,3000);
    function auto(){
        if(iNow==aLi.length-1){
            iNow=0;
        }
        else{
            iNow++;
        }
        for (var i = 0; i < aLi.length; i++) {
            thaiq.ui.fadeOut(aLi[i]);
        }
        thaiq.ui.fadeIn(aLi[iNow]);
    }
    function autoPrev(){
        if(iNow==0){
            iNow=aLi.length-1;
        }
        else{
            iNow--;
        }
        for (var i = 0; i < aLi.length; i++) {
            thaiq.ui.fadeOut(aLi[i]);
        }
        thaiq.ui.fadeIn(aLi[iNow]);
    }

    oPrevBg.onmouseover=oPrev.onmouseover=function(){
        oPrev.style.display='block';
        clearInterval(timer);
    };
    oNextBg.onmouseover=oNext.onmouseover=function(){
        oNext.style.display='block';
        clearInterval(timer);
    }
    oPrevBg.onmouseout=oPrev.onmouseout=function(){
        oPrev.style.display='none';
        timer=setInterval(auto,3000);
    };
    oNextBg.onmouseout=oNext.onmouseout=function(){
        oNext.style.display='none';
        timer=setInterval(auto,3000);
    }
    oPrev.onclick=function(){
        autoPrev();
    }
    oNext.onclick=function(){
        auto();
    }
}`

3.无缝切换

    `thaiq.app.toRun=function(){
    var oRun=document.getElementById('run1');
    var oUl=oRun.getElementsByTagName('ul')[0];
    var aLi=oUl.getElementsByTagName('li');

    var oPrev=thaiq.tools.getByClass(oRun,'prev')[0];
    var oNext=thaiq.tools.getByClass(oRun,'next')[0];

    var iNow=0;

    oUl.innerHTML+=oUl.innerHTML;
    oUl.style.width=aLi.length*aLi[0].offsetWidth+'px';

    oPrev.onclick=function(){
        if(iNow==0){
            iNow=aLi.length/2;
            oUl.style.left=-oUl.offsetWidth/2+'px';
        }
        thaiq.ui.moverLeft(oUl,-iNew*aLi[0].offsetWidth,-(iNow-1)*aLi[0].offsetWidth);
        iNow--;
    };
    oNext.onclick=function(){
        if(iNow==aLi.length/2){
            iNow=0;
            oUl.style.left=0;
        }
        thaiq.ui.moverLeft(oUl,-iNew*aLi[0].offsetWidth,-(iNow+1)*aLi[0].offsetWidth);
        iNow++;
    };
};`
